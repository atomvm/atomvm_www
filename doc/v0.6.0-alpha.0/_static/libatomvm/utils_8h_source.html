190/*
191 * The following is a workaround for disabling
192 * "warning: ISO C forbids conversion of function pointer to object pointer type [-Wpedantic]".
193 * It also makes use of _Static_assert to actually check if it is safe or not.
194 */
195
196typedef void (*func_ptr_t)(void);

198#ifdef __GNUC__
199
200static inline __attribute__((always_inline)) void *cast_func_to_void_ptr(func_ptr_t func)
201{
202#if __STDC_VERSION__ >= 201112L
203 _Static_assert(sizeof(void *) >= sizeof(func_ptr_t), "function ptr cannot be casted to void *");
204#endif
205
206/*
207 The following workaround is way more standard, but overcomplicated:
208
209 union {
210 func_ptr_t fp;
211 void *vp;
212 } u;
213
214 u.fp = func;
215
216 return u.vp;
217*/
218
219// Let's rather play with GCC diagnostic
220#pragma GCC diagnostic push
221#pragma GCC diagnostic ignored "-Wpedantic"
222 return (void *) func;
223#pragma GCC diagnostic pop
224}
